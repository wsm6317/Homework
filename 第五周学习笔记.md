## 1.归并算法的基本实现
复杂度为O(nlog n)

``` java
import java.util.Arrays;
public class MergeSort {

    private MergeSort(){}

    public static <E extends Comparable<E>> void sort(E[] arr){
        sort(arr,0,arr.length-1);
    }

    //通过递归将数组不断分为两份，将每份进行排序，最后两两组合起来，最终使得数组排序成功
    private static <E extends Comparable<E>> void sort(E[] arr,int l,int r){
        if(l>=r)
            return;
        //防止整型溢出
        int mid=l+(r-l)/2;
        //对arr[l,mid]进行排序
        sort(arr, l, mid);
        //对arr[mid+1,r]进行排序
        sort(arr,mid+1,r);
        merge(arr,l,mid,r);
    }

    //合并两个有序的区间arr[l,mid]和arr[mid+1,r]
    private static <E extends Comparable<E>> void merge(E[] arr,int l,int mid,int r){
        E[] temp= Arrays.copyOfRange(arr, l, r+l);
        int i=l,j=mid+1;

        //每轮循环为arr[k]赋值
        for (int k = l; k <=r ; k++) {
            if(i>mid)
            {arr[k]=temp[j-l];j++;}
            else if (j>r)
            {arr[k]=temp[i-l];i++;}
            else if(temp[i-l].compareTo(temp[j-l])<=0)
            {arr[k]=temp[i-l];i++;}
            else
            {arr[k]=temp[j-l];j++;}
        }
    }
}

```

//对arr[l,mid]进行排序
sort2(arr,l,mid);

sort2(arr,mid+1,r);

if(i>mid){
arr[k]=temp[j];
j++;
}
else if(j>r){
arr[k] =temp[i];
i++;
}

arr[k]=temp[i];
i++;
}

public class SortHelper {
    //将构造函数设为私有函数，使得类外部无法创建类对象
    private SortHelper(){}

    //性能测试函数，传入排序算法的名字和进行排序的数组
    public static <E extends Comparable<E>> void SortTest(String sortname,E[] arr){
        if(sortname.equals("SelectionSort"))
        {
            long StartTime = System.nanoTime();
            SelectionSort.sort(arr);
            if(!IsSorted(arr))
                //若没有成功排序，则抛出异常
                throw new RuntimeException(sortname+" failed");
            long EndTime = System.nanoTime();
            double time=(EndTime-StartTime)/1000000000.0;
            System.out.println(String.format("%s n=%d time=%f s",sortname,arr.length,time));
        }
        else if(sortname.equals("InsertSort"))
        {
            long StartTime = System.nanoTime();
            InsertSort.sort(arr);
            if(!IsSorted(arr))
                //若没有成功排序，则抛出异常
                throw new RuntimeException(sortname+" failed");
            long EndTime = System.nanoTime();
            double time=(EndTime-StartTime)/1000000000.0;
            System.out.println(String.format("%s n=%d time=%f s",sortname,arr.length,time));
        }
        else if(sortname.equals("MergeSort"))
        {
            long StartTime = System.nanoTime();
            MergeSort.sort(arr);
            if(!IsSorted(arr))
                //若没有成功排序，则抛出异常
                throw new RuntimeException(sortname+" failed");
            long EndTime = System.nanoTime();
            double time=(EndTime-StartTime)/1000000000.0;
            System.out.println(String.format("%s n=%d time=%f s",sortname,arr.length,time));
        }
    }

    //测试数组是否完成从小到大排序,若完成排序则返回1，未完成则返回0
    public static <E extends Comparable<E>> boolean IsSorted(E[] arr)
    {
        for(int i=1;i<arr.length;i++)
            if(arr[i-1].compareTo(arr[i])>0)
                return false;
        return true;
    }
}

import java.util.Random;
public class ArrayGenerator {
    //将构造函数设为私有函数，使得类外部无法创建类对象
    private ArrayGenerator(){}

    //生成有序数组，大小为n，从小到大排序
    public static Integer[] GenerateOrderedArray(int n)
    {
        Integer arr[]=new Integer[n];
        for(int i=0;i<n;i++)
            arr[i]=i;
        return arr;
    }

    //生成随机数组，大小为n，区间为(0,bound]
    public static Integer[] GenerateRandomArray(int n,int bound)
    {
        Integer arr[]=new Integer[n];
        Random r=new Random();
        for(int i=0;i<n;i++)
            arr[i]=r.nextInt(bound);
        return arr;
    }
}

class Solution { public  int[] twoSum(int[] nums, int target) { 
int [] answer=new  int[2]; 
for(int i=0;i<nums.length;i++) 
	for(int j=i+1;j<nums.length;j++)
		if(nums[i]+nums[j]==target) 
		{answer[0]=i;answer[1]=j;} 
		return answer; 
	} 
}
